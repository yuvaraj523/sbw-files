<script>
    // Function to get data from sessionStorage
    function getStoredEntries() {
        return JSON.parse(sessionStorage.getItem('entries')) || [];
    }

    // Function to save data to sessionStorage
    function saveEntry(entry) {
        const entries = getStoredEntries();
        entries.push(entry);
        sessionStorage.setItem('entries', JSON.stringify(entries));
    }

    // Function to display data from sessionStorage
    function displayEntries() {
        const entries = getStoredEntries();
        let table = document.getElementById('resultTable');
        if (!table) {
            table = document.createElement('table');
            table.id = 'resultTable';
            table.innerHTML = `
                <thead>
                    <tr>
                        <th>Name</th>
                        <th>Subject</th>
                        <th>Email</th>
                        <th>Phone</th>
                        <th>Message</th>
                        <th>Save Info</th>
                    </tr>
                </thead>
                <tbody></tbody>
            `;
            document.getElementById('result').appendChild(table);
        }

        // Clear existing rows
        table.querySelector('tbody').innerHTML = '';

        // Add rows from sessionStorage data
        entries.forEach(entry => {
            const newRow = table.querySelector('tbody').insertRow();
            newRow.innerHTML = `
                <td>${entry.name}</td>
                <td>${entry.subject}</td>
                <td>${entry.email}</td>
                <td>${entry.phone}</td>
                <td>${entry.message}</td>
                <td>${entry.saveInfo ? 'Yes' : 'No'}</td>
            `;
        });
    }

    document.getElementById('quoteForm').addEventListener('submit', function(event) {
        event.preventDefault(); // Prevent the form from submitting the traditional way

        // Retrieve values from the form
        const name = document.getElementById('name').value;
        const subject = document.getElementById('subject').value;
        const email = document.getElementById('email').value;
        const phone = document.getElementById('phone').value;
        const message = document.getElementById('message').value;
        const saveInfo = document.getElementById('save-info').checked;

        // Save the entry to sessionStorage
        const entry = { name, subject, email, phone, message, saveInfo };
        saveEntry(entry);

        // Display the updated table
        displayEntries();

        // Reset the form
        document.getElementById('quoteForm').reset();
    });

    // Display entries when the page loads
    window.onload = displayEntries;
</script>
